function bind(context, ...rest) {
    const self = this
    const empty = function() {};
    const fBound = function() {
        self.apply(this instanceof self ? this : context, rests.concat(Array.prototype.slice.call(arguments)))
    }
    empty.prototype = self.prototype;
    fBound.prototype = new empty();

    return fBound
}

# Promise.all

function PromiseAll(promises) {
    return new Promise((resolve, reject) => {
        const results = []
        const len = promises.length
        let count = 0
        promises.forEach((promise, i) => {
            promise.then((data) => {
                results[i] = data
                if (++count === len) {
                    return resolve(results)
                }
            }).catch((e) => {
                reject(e)
            })
        })
    })
}

# Array.flat

function flat(array, deep = 1) {
    if (deep === 0) return array.slice()
    return array.reduce((pre, cur) => {
        return pre.concat(Array.isArray(cur) ? flat(cur, deep - 1) : cur)
    }, [])
}

class MyPromise {
    constructor(executor) {
        this.status = 'pending'
        this.value = void 0
        this.reason = void 0
        this.fulfilledCallbacks = []
        this.rejectedCallbacks = []
        const resolve = (data) => {
            if (this.status === 'pending') {
                this.status = 'fulfilled'
                this.value = data
                this.fulfilledCallbacks.forEach(fn => fn())
            }
        }
        const reject = (e) => {
            if (this.status === 'pending') {
                this.status = 'rejected'
                this.reason = e
                this.rejectedCallbacks.forEach(fn => fn())
            }
        }
        try {
            executor(resolve, reject)
        } catch(e) {
            reject(e)
        }
    }
    then(onFulFilled, onRejected) {
        onFulFilled = typeof onFulFilled === 'function' ? onFulFilled : y => y
        onRejected = typeof onRejected === 'function' ? onRejected : err => throw err
        let promise2
        if (this.status === 'fulfilled') {
            promise2 = new Promise((resolve, reject) => {
                setTimeout(() => {
                    try {
                        const result = onFulFilled(this.data)
                        resolvePromise(promise2, result, resolve, reject)
                    } catch(e) {
                        reject(e)
                    }
                }, 0)
            })
        } else if (this.status === 'rejected') {
            promise2 = new Promise((resolve, reject) => {

                        const result = onRejected(this.reason)
                        resolvePromise(promise2, result, resolve, reject)
                    } catch(e) {
                        reject(e)
                    }
                }, 0)
            })
        } else if (this.status === 'pending') {
            promise2 = new Promise((resolve, reject) => {
                this.fulfilledCallbacks.push(() => {
                    setTimeout(() => {
                        try {
                            const result = onFulFilled(this.data)
                            resolvePromise(promise2, result, resolve, reject)
                        } catch(e) {

function resolvePromise(promise2, x, resolve, reject) {
    if (x === promise2) return new TypeError('循环引用')
    if (x != null && (typeof x === 'object' || typeof x === 'function')) {
        let called
        try {
            let then = x.then
            if (typeof then === 'function') {
                then.call(x, y => {
                    if (called) return
                    called = true
                    resolvePromise(promise2, y, resolve, reject)
                }, r => {
                    if (called) return
                    called = true
                    reject(r)
                })
            }
        } catch(e) {
            if (called) return
            called = true
            reject(e)
        }
    } else {
        resolve(x)
    }
}


class MyPromise {
  // 三种状态常量
  static PENDING = 'pending';
  static FULFILLED = 'fulfilled';
  static REJECTED = 'rejected';

  constructor(executor) {
    // 初始状态
    this.status = MyPromise.PENDING;
    // 成功值、失败原因
    this.value = null;
    this.reason = null;
    // 存放then回调队列（支持多次调用then）
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];

    // resolve 函数
    const resolve = (value) => {
      // 状态只能变更一次
      if (this.status !== MyPromise.PENDING) return;
      this.status = MyPromise.FULFILLED;
      this.value = value;
      // 异步执行所有成功回调
      queueMicrotask(() => {
        this.onFulfilledCallbacks.forEach(fn => fn());
      });
    };

    // reject 函数
    const reject = (reason) => {
      if (this.status !== MyPromise.PENDING) return;
      this.status = MyPromise.REJECTED;
      this.reason = reason;
      queueMicrotask(() => {
        this.onRejectedCallbacks.forEach(fn => fn());
      });
    };

    // 执行器报错直接走reject
    try {
      executor(resolve, reject);
    } catch (err) {
      reject(err);
    }
  }

  then(onFulfilled, onRejected) {
    // 穿透处理：不传回调则默认透传值/错误
    onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : val => val;
    onRejected = typeof onRejected === 'function' ? onRejected : err => { throw err };

    // 返回新Promise实现链式调用
    return new MyPromise((resolve, reject) => {
      // 封装成功回调
      const handleFulfilled = () => {
        try {
          const res = onFulfilled(this.value);
          resolve(res);
        } catch (err) {
          reject(err);
        }
      };
      // 封装失败回调
      const handleRejected = () => {
        try {
          const res = onRejected(this.reason);
          resolve(res);
        } catch (err) {
          reject(err);
        }
      };

      // 分三种状态处理
      if (this.status === MyPromise.FULFILLED) {
        queueMicrotask(handleFulfilled);
      } else if (this.status === MyPromise.REJECTED) {
        queueMicrotask(handleRejected);
      } else {
        // pending状态，存入队列等待
        this.onFulfilledCallbacks.push(handleFulfilled);
        this.onRejectedCallbacks.push(handleRejected);
      }
    });
  }

  // 简易catch，等价then(null, fn)
  catch(onRejected) {
    return this.then(null, onRejected);
  }

  // finally：无论成功失败都执行，不改变原有值
  finally(callback) {
    return this.then(
      val => MyPromise.resolve(callback()).then(() => val),
      err => MyPromise.resolve(callback()).then(() => { throw err })
    );
  }

  // 静态resolve
  static resolve(value) {
    // 如果是MyPromise实例直接返回
    if (value instanceof MyPromise) return value;
    return new MyPromise(resolve => resolve(value));
  }

  // 静态reject
  static reject(reason) {
    return new MyPromise((_, reject) => reject(reason));
  }
}

# reduce

function reduce(arr, fn, ...init) {
    let pre = init.length ? init[0] : arr[0]
    for (let i = init.length ? 1 : 0; i < arr.length; i++) {
        pre = fn(pre, arr[i], i)
    }
    return pre
}

# freeze

const _objectFreeze = object => {
    // 补全代码
    if(typeof object !== 'object' || object === null) {
        throw new TypeError(`the ${object} is not a object`)
    }

    const keys = Object.getOwnPropertyNames(object)
    const symbols = Object.getOwnPropertySymbols(object)

    [...keys, ...symbols].forEach(key => {
        Object.defineProperty(object, key, {
            configurable: false,
            writable: false,
        })
    })

    Object.preventExtensions(object)
}
